#include<iostream>
#include<cstring>
using namespace std;

int editDistance(string s1, string s2){

  /*  Don't write main().
   *  Don't read input, it is passed as function argument.
   *  Return output and don't print it.
   *  Taking input and printing output is handled automatically.
   */

    int m= s1.length();
    int n= s2.length();
    
    int **dp= new int*[m+1];
    for(int i=0;i<=m;i++){
        dp[i]= new int[n+1];
    }
    
    for(int i=0;i<=n;i++){
        dp[0][i]=i;
    }
    for(int i=0;i<=m;i++){
        dp[i][0]=i;
    }
    
    
    for(int i=1;i<=m;i++){
        for(int j=1;j<=n;j++){
            if(s1[i-1]==s2[j-1])
                dp[i][j]=dp[i-1][j-1];
            else{
                dp[i][j]=1+min(dp[i-1][j], min(dp[i-1][j-1], dp[i][j-1]));
            }
        }
    }
    return dp[m][n];

}
